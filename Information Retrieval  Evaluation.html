def counter(RIsys):
    recall=[]
    precision=[]
    count=0
    for i in RIsys :
        if i in REL:
            count=1+count
        else:
            count=0+count
        recall.append(count/len(REL))
        precision.append(count/len(recall))
        
    return recall,precision
